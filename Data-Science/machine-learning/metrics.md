# F1-Score
f = (1+\beta^2) * TP / [(1+\beta^2) TP + \beta^2 * FN + FP)]

f1 = 2 * recall*precision / (recall + precision)

# Matthew's Correlation Coefficient
MCC = (TP*TN - FP*FN)/[(TP+FP)(TP+FN)(TN+FP)(TN+FN)]
